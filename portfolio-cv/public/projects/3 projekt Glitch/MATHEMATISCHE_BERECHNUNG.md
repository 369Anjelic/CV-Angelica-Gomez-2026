## 💻 CODE-IMPLEMENTIERUNG

```javascript
// Frontend: Berechne Progress automatisch
function calculateProgress() {
    let totalTopics = 0;
    let learnedTopics = 0;
    
    Object.values(achievements).forEach(lf => {
        const topics = lf.topics || [];
        totalTopics += topics.length;
        learnedTopics += topics.filter(t => t.learned).length;
    });
    
    // Berechne Prozent mit Rounding
    const percentage = totalTopics > 0 
        ? Math.round((learnedTopics / totalTopics) * 100) 
        : 0;
    
    return {
        learned: learnedTopics,
        total: totalTopics,
        percentage: percentage
    };
}
```
